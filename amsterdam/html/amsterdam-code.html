The main idea is to separate each platform into control-plane logic and data-plane logic: 主要思想是将每个平台分为 control plane 逻辑和 data plane 逻辑：

  • The control plane decides intent, policy, routing, scheduling, versioning, rollout, and observability.

That split shows up in both projects. In Project 1, the control plane decides where each AI workflow should run and how it should recover from failure. Core Reasoning Model

Across both projects, the same reasoning model applies: 在这两个项目中，适用相同的推理模型：

将policy与执行分开。

  • Separate orchestration from inference.

将orchestration与推理分开。

  • Separate retrieval from ranking.

将 retrieval 与 ranking 分开。

  • Separate online serving from offline or nearline processing.

将 online 服务与 offline 或 nearline 处理分开。

Consumer pattern: 消费 schema：

async def consume_ready_steps(consumer, state_store, executor):
    async for event in consumer:
        if await state_store.is_step_completed(event.workflow_id, event.step_id):
            continue

        lease = await state_store.acquire_step_lease(
            event.workflow_id,
            event.step_id,
            ttl_s=120,
        )
        if not lease.acquired:
            continue

        try:
            result = await executor.execute(event)
            await state_store.complete_step(event.workflow_id, event.step_id, result)
            await publish_step_completed(event, result)

            await state_store.release_lease(event.workflow_id, event.step_id)
            await publish_capacity_waiting(event)

Resource Controller Implementation

资源控制器实现

The resource controller has two loops: 资源控制器有两个循环：

  • Short loop: observe queue/latency/token/GPU metrics and adjust replicas.

短循环：观察队列/latency/token/GPU metrics 并调整副本。

  • Long loop: optimize placement, warm models, and bin-pack workloads.

长循环：优化放置、热模型和装箱 workload。

class ResourceController:
    def __init__(self, metrics, scheduler, scaler, policy):
        self.metrics = metrics
        self.scheduler = scheduler
        self.scaler = scaler
        self.policy = policy

    async def reconcile(self):
        pools = await self.metrics.get_pool_snapshots()
        for pool in pools:
            desired = self._desired_replicas(pool)
            await self.scaler.scale(pool.name, desired)

        placements = await self.scheduler.current_placements()
        recommendations = self._binpack(placements, pools)
        await self.scheduler.apply(recommendations)

    def _desired_replicas(self, pool):
        if pool.kind == "vllm":
            pressure = max(
                pool.waiting_tokens / self.policy.target_waiting_tokens,
                pool.ttft_p95_ms / self.policy.target_ttft_ms,
                pool.kv_cache_utilization / self.policy.target_kv_cache_util,
            )
        elif pool.kind == "triton":
            pressure = max(
                pool.queue_ms_p95 / self.policy.target_queue_ms,
                pool.request_latency_p95_ms / self.policy.target_latency_ms,
            )
        else:
            pressure = pool.queue_depth / self.policy.target_queue_depth

        return clamp(
            round(pool.current_replicas * pressure),
            pool.min_replicas,
            pool.max_replicas,
        )

def reciprocal_rank_fusion(candidate_lists, k=60):
    scores = defaultdict(float)
    for candidates in candidate_lists:
        for rank, candidate in enumerate(candidates, start=1):
            scores[candidate.asset_id] += 1.0 / (k + rank)
    return sorted(scores.items(), key=lambda x: x[1], reverse=True)

def mmr_rerank(candidates, relevance, similarity, lambda_relevance=0.75, k=40):
    selected = []
    remaining = set(candidates)

    while remaining and len(selected) < k:
        best = None
        best_score = float("-inf")
        for candidate in remaining:
            diversity_penalty = 0
            if selected:
                diversity_penalty = max(similarity(candidate, s) for s in selected)
            score = (
                lambda_relevance * relevance(candidate)
                - (1 - lambda_relevance) * diversity_penalty
            )
            if score > best_score:
                best = candidate
                best_score = score
        selected.append(best)
        remaining.remove(best)

    return selected

class IdempotencyGuard:
    def __init__(self, redis_client, workflow_store, ttl_seconds=900):
        self.redis = redis_client
        self.workflow_store = workflow_store
        self.ttl = ttl_seconds

    async def acquire_or_get(self, tenant_id: str, intent_id: str, key: str):
        redis_key = f"idemp:{tenant_id}:{intent_id}:{key}"
        existing = await self.redis.get(redis_key)
        if existing:
            return {"status": "duplicate", "workflow_id": existing.decode("utf-8")}

        workflow = await self.workflow_store.find_by_idempotency_key(
            tenant_id=tenant_id,
            intent_id=intent_id,
            idempotency_key=key,
        )
        if workflow:
            await self.redis.set(redis_key, workflow.workflow_id, ex=self.ttl)
            return {"status": "duplicate", "workflow_id": workflow.workflow_id}

        lock_key = f"lock:{redis_key}"
        acquired = await self.redis.set(lock_key, "1", nx=True, ex=30)
        if not acquired:
            return {"status": "in_flight"}
        return {"status": "new", "lock_key": lock_key}

Circuit breaker logic: circuit breaker逻辑：

class CircuitBreaker:
    def __init__(self, failure_threshold: int, open_seconds: int):
        self.failure_threshold = failure_threshold
        self.open_seconds = open_seconds
        self.failures = 0
        self.open_until_ms = 0

    def allow(self, now_ms: int) -> bool:
        return now_ms >= self.open_until_ms

    def record_success(self):
        self.failures = 0

    def record_failure(self, now_ms: int):
        self.failures += 1
        if self.failures >= self.failure_threshold:
            self.open_until_ms = now_ms + self.open_seconds * 1000


async def call_with_breaker(name, breaker, now_ms, call, fallback):
    if not breaker.allow(now_ms):
        return await fallback(reason=f"{name}_circuit_open")
    try:
        result = await call()
        breaker.record_success()
        return result
    except TimeoutError:
        breaker.record_failure(now_ms)
        return await fallback(reason=f"{name}_timeout")

def binpack_score(profile: ModelProfile, node: NodeCapacity) -> float:
    fragmentation_penalty = 0.0
    if profile.mig_compatible and node.free_full_gpus > 0:
        fragmentation_penalty += 0.2
    if profile.runtime == "vllm" and node.free_full_gpus == 0:
        fragmentation_penalty += 10.0

    latency_penalty = max(0, node.current_queue_ms - latency_budget_ms(profile.latency_class)) / 1000
    locality_bonus = 0.1 if node.reserved_for_interactive and profile.latency_class == "interactive" else 0
    return fragmentation_penalty + latency_penalty - locality_bonus

Graceful degradation: 优雅降级：

async def search_with_degradation(query, ctx):
    vector_task = asyncio.create_task(vector_retrieve(query, ctx))
    lexical_task = asyncio.create_task(lexical_retrieve(query, ctx))
    graph_task = asyncio.create_task(graph_retrieve(query, ctx))

    done, pending = await asyncio.wait(
        [vector_task, lexical_task, graph_task],
        timeout=ctx.latency_budget_ms / 1000,
        return_when=asyncio.ALL_COMPLETED,
    )

    for task in pending:
        task.cancel()

    candidates = merge_successful_sources(done)
    if not candidates:
        candidates = await lexical_fallback(query, ctx)

    rerank_depth = 200 if ctx.remaining_budget_ms > 80 else 50
    features = await join_ranking_features(candidates[:rerank_depth], ctx.user_context, feature_store, cache)
    return rank(candidates, features, ctx)
